git config --global user.name ""  // Имя
git config --global user.email (почта) // Почта

cd (Путь) // Перемещение по директории
git init // Сделать папку репозиторием
git remote add origin (url репозитория) // Связать локалку с удалённым
git clone (url репозитория) // Стянуть с нуля

git pull // Стянуть с удалённой

git status // Посмотреть чего нет в коммите
git add // Добавить файл к коммиту
git commit -m "(Месседж)" // Создать коммит
git push // Отослать
git log // Посмотреть историю Коммитов

git checkout (id) // Перемещение по коммитам/веткам

git branch // Создать или посмотреть ветку(и)
git merge (Имя ветки) // Соединить ветки
